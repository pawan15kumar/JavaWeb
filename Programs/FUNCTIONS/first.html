class Main{
            static int minOfThree(int x, int y, int z){
                if(x < y){
                    if(x < z){
                        return x;
                    }
                    else{
                        return z;
                    }
                }
                else{
                    if(y < z){
                        return y;
                    }
                    else{
                        return z;
                    }
                }
            }
            public static void main(String[] args){	
                System.out.println("Minimum value of 105, 9, 209 is " + minOfThree(105,9,209));
            }
        }
          Output
        Minimum value of 105, 9, 209 is 9
