class Main{
            static Boolean isPrime(int num){
                for(int i = 2; i < num; i++){
                    if(num % i == 0){
                        return false;
                    }
                }
                return true;
            }
            static void twinPrime(){
                int prevPrime = 0;
                for(int i = 3; i <= 100; i++){
                    if(isPrime(i)){
                        if((i - prevPrime) == 2){
                            System.out.println(prevPrime + " and " + i + " are twin Prime");
                        }
                        prevPrime = i;
                    }
                }
            }
            public static void main(String[] args){
                twinPrime();
            }
        }
