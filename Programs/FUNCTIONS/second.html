class Main{
            static double avgOfThree(int x, int y, int z){
                return ((x + y + z)/3.0);
            }
            public static void main(String[] args){	
                System.out.println("Average of 105, 9, 209 is " + avgOfThree(105,9,209));
            }
        }
          Output
        Average of 105, 9, 209 is 107.66666666666667
