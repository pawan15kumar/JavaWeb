import java.sql.*;
        import java.io.*;
        
        class first {
            public static void main(String x[]) throws ClassNotFoundException, SQLException, IOException {
                Class.forName("com.mysql.cj.jdbc.Driver");
                Connection c = DriverManager.getConnection("jdbc:mysql://localhost:3306/student", "root", "root");// throws sql
                                                                                                                  // exception
                Statement st = c.createStatement(ResultSet.TYPE_SCROLL_INSENSITIVE, ResultSet.CONCUR_READ_ONLY);
                InputStreamReader isr = new InputStreamReader(System.in);
                BufferedReader br = new BufferedReader(isr);
                System.out.println("Enter the row number:");
                int row = Integer.parseInt(br.readLine());
                ResultSet rs = st.executeQuery("Select * from stu");
                if (rs.absolute(row)) {
                    System.out.println("Row " + row + ":");
                    System.out.println("Column 1: " + rs.getString(1));
                    System.out.println("Column 2: " + rs.getInt(2));
        
                } else {
                    System.out.println("Row number " + row + " does not exist.");
                }
                c.close();
            }
        }
