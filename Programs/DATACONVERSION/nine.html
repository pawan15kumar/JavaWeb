class StringToDate{
            public static void main(String st[]) throws java.text.ParseException
            {
                String str = "2024-01-01";
                java.time.LocalDate date = java.time.LocalDate.parse(str);
                System.out.println("Converted Date : "+date);
            }
        }

    Converted Date : 2024-01-01
