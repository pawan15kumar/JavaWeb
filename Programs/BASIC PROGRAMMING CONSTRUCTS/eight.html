class CompareNum{
            public static void main(String st[]){
                int num1 = 10;
                int num2 = 20;
                if(num1>num2){
                    System.out.println(num1 + " is greater than "+num2);
                }
                else if(num1 < num2){
                    System.out.println(num1 + " is less than "+num2);
                }
                else{
                    System.out.println(num1 + " is equal to "+num2);
                }
            }
        }
    
    Output:
    
    10 is less than 20
