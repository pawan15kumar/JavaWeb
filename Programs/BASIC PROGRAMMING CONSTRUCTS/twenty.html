class Main{
            static Boolean isPrime(int num){
                int i = 0;
                if(num == 2)
                    return true;
                for(i = 2; i < num; i++){
                    if(num%i == 0){
                        return false;
                    }
                }
                if(i == num)
                    return true;
                else
                    return false;
            }
            public static void main(String[] args){
                int sum = 0;
                int count = 0;
                for(int i = 2; count < 100; i++){
                    if(isPrime(i)){
                        sum += i;
                        count++;
                    }
                }
                System.out.println("Sum of first 100 Prime Numbers : " + sum);
            }
        }
          Output
        Sum of first 100 Prime Numbers : 24133
