class AddDigits{
            public static void main(String st[]){
                int num = 345;
                int sum=0;
                while(num != 0){
                    sum += num % 10;
                    num /= 10;
                }
                
                System.out.println("Sum of digits : "+ sum);	
            }
        }
    Output:
    
    Sum of digits : 12
