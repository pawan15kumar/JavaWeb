class Product{
            private int num1, num2;
            Product(int N1, int N2){
                num1 = N1;
                num2 = N2;
                System.out.println("Value of num1 = "+num1);
                System.out.println("Value of num2 = "+num2);    
            }
            int multi(){
                return num1 * num2;
            }
        }
        
    class Main{
            public static void main(String args[]){
                Product product = new Product(2,3);
                System.out.println("The Product of num1 and num2 = "+product.multi());
            }
        }
    Output:
    
    Value of num1 = 2
    Value of num2 = 3
    The Product of num1 and num2 = 6
