import java.util.Scanner;
        
        class sum {
            public static void main(String[] args) {
                int sum = 0;
                Scanner obj = new Scanner(System.in);
                System.out.print("How many numbers you want to print:");
                int n = obj.nextInt();
                for (int i = 1; i <= n; i++) {
                    sum = sum + i;
        
                }
                System.out.print("The sum of " + n + " numbers is " + sum);
        
                obj.close();
            }
        }
