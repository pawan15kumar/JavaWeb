import java.io.*;
        
        class A {
            public static void main(String args[]) throws java.io.IOException {
                InputStreamReader isr = new InputStreamReader(System.in);
                BufferedReader obj = new BufferedReader(isr);
                System.out.print("Enter a number:");
                int num1 = Integer.parseInt(obj.readLine());
                switch (num1) {
                    case 1:
                        System.out.println("Hey it's Sunday!");
                        break;
                    case 2:
                        System.out.println("Hey it's Monday!");
                        break;
                    case 3:
                        System.out.println("Hey it's Tuesday!");
                        break;
                    case 4:
                        System.out.println("Hey it's Wednesday!");
                        break;
                    case 5:
                        System.out.println("Hey it's Thursday!");
                        break;
                    case 6:
                        System.out.println("Hey it's Friday!");
                        break;
                    case 7:
                        System.out.println("Hey it's Saturday!");
                        break;
        
                    default:
                        System.out.println("Enter a valid number for days");
        
                }
            }
        }
