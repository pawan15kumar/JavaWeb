class A {
            static int x[] = { 1, 2, 3, 4, 5 };
        
            public static void main(String args[]) {
                int first = x[0];
                int last = x[(x.length - 1)];
                System.out.println("Original array");
                for (int i = 0; i < x.length; i++)
                    System.out.print(x[i] + " ");
                x[0] = last;
                x[(x.length - 1)] = first;
                System.out.println("\n Changed array");
                for (int i = 0; i < x.length; i++)
                    System.out.print(x[i] + " ");
            }
        }
