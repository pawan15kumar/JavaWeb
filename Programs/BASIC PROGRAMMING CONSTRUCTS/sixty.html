import java.util.Scanner;
        
        class Cube {
            public static void main(String arg[]) {
                Scanner obj = new Scanner(System.in);
                System.out.println("Enter the number: ");
                int n = obj.nextInt();
                for (int i = 0; i <= n; i++) {
                    System.out.println("Cube of " + i + " is: " + i * i * i);
                }
            }
        }
