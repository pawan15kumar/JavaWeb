class A {
            int product(int a, int b) {
                return a * b;
            }
        
            int sum(int a, int b) {
                return a + b;
            }
        
            int subtract(int a, int b) {
                return a - b;
            }
        
            int remainder(int a, int b) {
                return a % b;
            }
        
            int divide(int a, int b) {
                return a / b;
            }
        
            public static void main(String x[]) {
                A obj = new A();
                System.out.println("Sum of two numbers" + "   " + obj.sum(14, 31));
                System.out.println("Difference of two numbers" + "   " + obj.subtract(14, 31));
                System.out.println("Product of two numbers" + "   " + obj.product(14, 31));
                System.out.println("Remainder of two numbers" + "   " + obj.remainder(14, 31));
                System.out.println("Division of two numbers" + "   " + obj.divide(14, 31));
        
            }
        }
