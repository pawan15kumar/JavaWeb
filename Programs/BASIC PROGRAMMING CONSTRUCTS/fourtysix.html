import java.io.*;
        
        class CGPACalculator {
        
            public static void main(String[] args) throws java.io.IOException {
                BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        
                System.out.print("Enter the number of subjects: ");
                int numSubjects = Integer.parseInt(br.readLine());
        
                double[] marksObtained = new double[numSubjects];
                double[] maxMarks = new double[numSubjects];
                double[] cgpaValues = new double[numSubjects];
        
                for (int i = 0; i < numSubjects; i++) {
                    System.out.print("Enter marks obtained in subject " + (i + 1) + ": ");
                    marksObtained[i] = Integer.parseInt(br.readLine());
                    System.out.print("Enter maximum marks for subject " + (i + 1) + ": ");
                    maxMarks[i] = Integer.parseInt(br.readLine());
                }
        
                for (int i = 0; i < numSubjects; i++) {
                    cgpaValues[i] = (marksObtained[i] / maxMarks[i]) * 10;
                }
        
                double totalCGPA = 0;
                for (int i = 0; i < numSubjects; i++) {
                    totalCGPA += cgpaValues[i];
                }
                double averageCGPA = totalCGPA / numSubjects;
        
                double percentage = averageCGPA * 10;
        
                System.out.println("Average CGPA: " + averageCGPA);
                System.out.println("CGPA Percentage: " + percentage + "%");
        
            }
        }
