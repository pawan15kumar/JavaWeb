class A {
            static int a[] = { 1, 2, 3, 4, 5, 6 };
        
            public static void main(String arg[]) {
                int countodd = 0;
                int counteven = 0;
                for (int i = 0; i < a.length; i++) {
                    if (a[i] % 2 == 0)
                        counteven++;
                    else
                        countodd++;
                }
                System.out.println("Odd no:" + countodd + " Even no:" + counteven);
            }
        }
