class A {
            static int largest(int arr[], int pos) {
                return arr[pos];
            }
        
            static int removeDuplicate(int arr[]) {
                int j = 0, i;
                for (i = 0; i < (arr.length - 1); i++) {
                    if (arr[i] != arr[i + 1]) {
                        arr[j++] = arr[i];
                    }
                }
                arr[j++] = arr[i];
                return j;
            }
        
            static void sort(int arr[]) {
                int k, i, j;
                for (i = 0; i < arr.length - 1; i++) {
                    for (j = i + 1; j < arr.length; j++) {
                        if (arr[i] > arr[j]) {
                            k = arr[i];
                            arr[i] = arr[j];
                            arr[j] = k;
                        }
                    }
                }
            }
        
            static void display(int arr[], int size) {
                for (int i = 0; i < size; i++) {
                    System.out.print(arr[i] + " ");
                }
            }
        
            public static void main(String args[]) {
                int[] array = { 1, 3, 2, 3, 4, 2, 1, 3, 1 };
                System.out.println("The original array:");
                display(array, array.length);
                sort(array);
                int size = removeDuplicate(array);
                // display(array,size);
                int k = largest(array,size-2);
                System.out.println("\nThe 2nd largest element in an array: " + k);
        
            }
        }
