import java.util.Scanner;
        
        class sum {
            public static void main(String[] args) {
                int fact = 1;
                Scanner obj = new Scanner(System.in);
                System.out.print("Enter the number:");
                int n = obj.nextInt();
                for (int i = 1; i <= n; i++) {
                    fact = fact * i;
        
                }
                System.out.print("The sum of " + n + " numbers is " + fact);
        
                obj.close();
            }
        }
