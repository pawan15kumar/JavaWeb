import java.io.*;
        
        class A {
            public static void main(String x[]) throws java.io.IOException {
                int op = 1, num1 = 9, num2 = 0;
                while (op != 5) {
                    System.out.println("Enter the operation you want to perform:" + "\n1.Addition" + "\n2.Subtraction"
                            + "\n3.Multiplication" + "\n4.Division" + "\n5.Exit");
                    InputStreamReader isr = new InputStreamReader(System.in);
                    BufferedReader obj = new BufferedReader(isr);
        
                    op = Integer.parseInt(obj.readLine());
                    if (op != 5) {
                        System.out.println("Enter first number:");
                        num1 = Integer.parseInt(obj.readLine());
        
                        System.out.println("Enter second number:");
                        num2 = Integer.parseInt(obj.readLine());
        
                    }
        
                    if (op == 1)
                        System.out.println("Sum of entered numbers is:" + (num1 + num2));
                    else if (op == 2)
                        System.out.println("Difference of entered numbers is:" + (num1 - num2));
        
                    else if (op == 3)
                        System.out.println("Product of entered numbers is:" + (num1 * num2));
        
                    else if (op == 4)
                        System.out.println("division of entered numbers is:" + (num1 / num2));
                    else if (op == 5)
                        System.out.println("Exiting");
        
                    else
                        System.out.println("Enter valid numbers");
                }
            }
        }
