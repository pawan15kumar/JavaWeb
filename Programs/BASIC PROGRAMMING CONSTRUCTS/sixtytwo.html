import java.util.Scanner;
        
        class Table {
            public static void main(String arg[]) {
                Scanner obj = new Scanner(System.in);
                System.out.println("Enter the number: ");
                int n = obj.nextInt();
                for (int i = 0; i <= 10; i++) {
                    System.out.println(n + " * " + i + " = " + n * i);
                }
            }
        }
