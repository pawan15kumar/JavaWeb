import java.io.*;

class Matrix {

    public static void main(String arg[]) throws java.io.IOException{
       InputStreamReader isr=new InputStreamReader(System.in);
       BufferedReader br =new BufferedReader(isr);

        System.out.print("Enter the number of rows: ");
        int rows = Integer.parseInt(br.readLine());
        System.out.print("Enter the number of columns: ");
        int columns =  Integer.parseInt(br.readLine());

        int matrix[][] = new int[rows][columns];
        

        System.out.println("Enter the elements of the first matrix:");
        for (int i = 0; i < rows; i++) {
            System.out.println("Enter elements for row "+ (i + 1));
            for (int j = 0; j < columns; j++) {
                matrix[i][j] = Integer.parseInt(br.readLine());
            }
        }

       System.out.print("Enter the number of rows for second matrix: ");
        int rows2 = Integer.parseInt(br.readLine());
        System.out.print("Enter the number of columns for second matrix: ");
        int columns2 =  Integer.parseInt(br.readLine());
        int matrix2[][] = new int[rows2][columns2];

        System.out.println("Enter the elements of the second matrix:");
        for (int i = 0; i < rows2; i++) {
            System.out.println("Enter elements for row "+ (i + 1));
            for (int j = 0; j < columns2; j++) {
                matrix2[i][j] = Integer.parseInt(br.readLine());
            }
        }

        
       int result[][] = new int[rows][columns2];

        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < columns2; j++) {
                result[i][j] = 0;
                for (int k = 0; k < columns; k++) {
                    result[i][j] = result[i][j] + (matrix[i][k] * matrix2[k][j]);
                }
            }
        }
       System.out.println("Multiplication of two Matrices:");
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < columns; j++) {
                System.out.print(result[i][j]+" ");
            } 
            System.out.println();
        }
       
        
    }
}
