class Animal{
            void move(){
                System.out.println("Animals Run fast");
            }
        }
        
        class Cheetah extends Animal{
            void move(){
                System.out.println("Run");
            }
        }
        
        class Main{
            public static void main(String args[]){
                Cheetah obj = new Cheetah();
                obj.move();
            }
        }
        OUTPUT

        Run
