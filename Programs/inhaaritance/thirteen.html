class Parent{
            void Worship(){
                System.out.println("Worship in the morning");
            }
        }
        
        class Child extends Parent{
            void Worship(){
                System.out.println("Worship in the Evening");
            }
        }
        
        class Main{
            public static void main(String args[]){
                Child obj = new Child();
                obj.Worship();
            }
        }
        OUTPUT
        Worship in the Evening
