abstract class Employee{
            abstract double calculateSalary();
            abstract void displayInfo();
        }
        
        class Manager extends Employee{
        private String Name;
        private double Salary, Bonus;
            Manager(String name, double BaseSal, double bonus){
                Name = name;
                Salary = BaseSal;
                Bonus = bonus;	
                
            }
            double calculateSalary(){
                    return Salary+Bonus;	
            }
            void displayInfo(){
                System.out.println("Manager's name : "+Name);
                System.out.println("The Base Salary of Manager is "+Salary);
                System.out.println("The provided Bonus of Manager is "+Bonus);
                System.out.println("Salary after Bonus : "+calculateSalary());
                
            }
        }
        
        class Programmer extends Employee{
        private String Name;
        private double Salary, Bonus;
            Programmer(String name, double BaseSal, double bonus){
                Name = name;
                Salary = BaseSal;
                Bonus = bonus;	
                
            }
            double calculateSalary(){
                    return Salary+Bonus;	
            }
            void displayInfo(){
                System.out.println("Programmer's name : "+Name);
                System.out.println("The Base Salary of Programmer is "+Salary);
                System.out.println("The provided Bonus of Programmer is "+Bonus);
                System.out.println("Salary after Bonus : "+calculateSalary());
                
            }
        
        
        }
        
        class Main{
            public static void main(String args[]){
                Manager manager = new Manager("pawan", 50000, 10000);
                    manager.displayInfo();
                Programmer programmer = new Programmer("Afseen", 40000, 2500);
                    programmer. displayInfo();
            }
        }
