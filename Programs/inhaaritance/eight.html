class Shape{
        double getPerimeter(){
            System.out.println("Perimeter of Circle is 2*3.14*r");
            return 0;
        }
        double getArea(){
            System.out.println("Area of Circle is 3.14*r*r");
            return 0;
        }
    
    }
    
    class Circle extends Shape{
    double Radius;
        Circle(double r){
            Radius = r;
        }
        double getPerimeter(){
            return 2 * 3.14 * Radius;
        }
        double getArea(){
            return 3.14 * (Radius*Radius);
        }
    }
    
    class Main{
        public static void main(String args[]){
            Circle obj = new Circle(10);
            System.out.println("The Perimeter of circle is "+obj.getPerimeter());
            System.out.println("The Area of circle is "+obj.getArea());
        }
    }
