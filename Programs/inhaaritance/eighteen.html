abstract class Animal{
            abstract void eat();
            abstract void sleep();
        }
        
        class Lion extends Animal{
            void eat(){
                System.out.println("Lions are Carnivores, eat large hoofed animals.");
            }
            void sleep(){
                System.out.println("Lion Sleeps 15-20 hours a day.");
            }
        }
        
        class Tiger extends Animal{
            void eat(){
                System.out.println("Tigers are apex predators, hunt large ungulates.");
            }
            void sleep(){
                System.out.println("Tiger sleeps 18-20 hours a day.");
            }
        }
        
        class Deer extends Animal{
            void eat(){
                System.out.println("Deer eat Grasses and sedges.");
            }
            void sleep(){
                System.out.println("Deer sleep an average 4.5 hours a day.");
            }
        }	
        
        class Main{
            public static void main(String args[]){
                Lion lion = new Lion();
                    lion.eat();
                    lion.sleep();
                Tiger tiger = new Tiger();
                    tiger.eat();
                    tiger.sleep();
                Deer deer = new Deer();
                    deer.eat();
                    deer.sleep();
            }
        }
        
        
OUTPUT
Lions are Carnivores, eat large hoofed animals.
Lion Sleeps 15-20 hours a day.
Tigers are apex predators, hunt large ungulates.
Tiger sleeps 18-20 hours a day.
Deer eat Grasses and sedges.
Deer sleep an average 4.5 hours a day.
