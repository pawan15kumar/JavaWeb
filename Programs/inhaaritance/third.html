class Shape{
            int getArea(){
                return 0; 
            }
        }
        
        class Rectangle extends Shape{
            int length, width;
            Rectangle(int L, int W){
                length = L;
                width = W;
            }
            int getArea(){
                return length * width;
            }
        }
        
        class Main{
            public static void main(String args[]){
                Rectangle obj = new Rectangle(5,3);
                System.out.println("Area of rectangle = "+obj.getArea());
            }
        }
        OUTPUT
        area of rectangle=15
