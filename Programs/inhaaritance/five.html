class BankAccount{
            double balance;
           
           BankAccount(double initialBalance){	
               balance = initialBalance;
           }
           void deposit(double amount){
               balance +=amount;
               System.out.println("Deposited " +amount+" new balance " +balance);
       
           }
       
           void withdraw(double amount){
                   balance -= amount;
               System.out.println("Withdrawn amount  "+amount+ " new balance  "+balance);	
           }
       }
       
       class SavingAccount extends BankAccount{
           SavingAccount(double initialBalance){
               super(initialBalance);
           }
           void withdraw(double amount){
               if(balance - amount<100){
                   System.out.println("Don't have sufficeint Amount to withdraw");
               }
               else{
                   super.withdraw(amount);
               }	
           }
       
       }
       
       class Main{
       public static void main(String args[]){
           SavingAccount obj = new SavingAccount(1000);
           obj.deposit(500);
           obj.withdraw(200);
           obj.withdraw(1500);
       }
       }
outout

Withdrawn amount  200.0 new balance  1300.0
