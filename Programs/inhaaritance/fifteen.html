abstract class Animal{
            abstract void sound();
        }
        
        class Lion extends Animal{
            void sound(){
                System.out.println("Lion Roar");
            }
            
        }
        
        class Tiger extends Animal{
            void sound(){
                System.out.println("Tiger Growl");
            }
        }
        
        class Main{
            public static void main(String args[]){
                Lion obj = new Lion();
                obj.sound();
                Tiger obj1 = new Tiger();
                obj1.sound();
            }
        }
        OUTPUT
        Lion Roar
Tiger Growl
